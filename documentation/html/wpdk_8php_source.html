187  /* Only backend administration */
188  if ( !is_admin() ) {

191

193  if ( empty( $transient->checked ) ) {
194  return $transient;

196 
197  if ( isset( $transient->response ) ) {
198  $plugins = get_plugins();
199  foreach ( $transient->response as $key => $value ) {
200  if ( isset( $plugins[$key] ) ) {

203  /* Have my package? */

205  unset( $transient->response[$key] );
